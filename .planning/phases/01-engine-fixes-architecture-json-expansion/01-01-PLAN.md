   CRITICAL: This MUST be `static` at module level, NOT inside the function body. `include_dir!` generates a `Dir<'static>` type that requires a `'static` binding.

4. Replace the entire `builtin_audits()` function body (lines 30-49) with:
   ```rust
   fn builtin_audits() -> Vec<JsonAuditFile> {
       BUILTIN_AUDITS_DIR
           .files()
           .filter(|f| f.path().extension().and_then(|e| e.to_str()) == Some("json"))
           .filter_map(|f| {
               let src = match f.contents_utf8() {
                   Some(s) => s,
                   None => {
                       eprintln!("Warning: built-in audit file {:?} is not valid UTF-8", f.path());
                       return None;
                   }
               };
               match serde_json::from_str::<JsonAuditFile>(src) {
                   Ok(audit) => Some(audit),
                   Err(e) => {
                       eprintln!("Warning: failed to parse built-in audit {:?}: {e}", f.path());
                       None
                   }
               }
           })
           .collect()
   }
   ```

5. Update the test `test_builtin_audits_returns_four` (line 123-133): Change the assertion from `assert_eq!(audits.len(), 4, ...)` to:
   ```rust
   assert!(audits.len() >= 4, "Expected at least 4 built-in audits, got {}", audits.len());
   ```
   This allows the test to pass both before and after the 36 per-language files are added.

6. Leave `test_builtin_audit_pipeline_names` (lines 136-143) unchanged — the four existing pipeline names will still be present in the 4 template files (until Plan 04 deletes them).
